266static inline CborError validate_utf8_string(const void *ptr, size_t n)
267{
268 const uint8_t *buffer = (const uint8_t *)ptr;
269 const uint8_t * const end = buffer + n;
270 while (buffer < end) {
271 uint32_t uc = get_utf8(&buffer, end);
272 if (uc == ~0U)
273 return CborErrorInvalidUtf8TextString;
274 }
275 return CborNoError;
276}

278static inline CborError validate_simple_type(uint8_t simple_type, uint32_t flags)
279{
280 /* At current time, all known simple types are those from RFC 7049,
281 * which are parsed by the parser into different CBOR types.
282 * That means that if we've got here, the type is unknown */
283 if (simple_type < 32)
284 return (flags & CborValidateNoUnknownSimpleTypesSA) ? CborErrorUnknownSimpleType : CborNoError;
285 return (flags & CborValidateNoUnknownSimpleTypes) == CborValidateNoUnknownSimpleTypes ?
286 CborErrorUnknownSimpleType : CborNoError;
287}

289static inline CborError validate_number(const CborValue *it, CborType type, uint32_t flags)
290{
291 CborError err = CborNoError;
292 size_t bytesUsed, bytesNeeded;
293 uint64_t value;
294
295 if ((flags & CborValidateShortestIntegrals) == 0)
296 return err;
297 if (type >= CborHalfFloatType && type <= CborDoubleType)
298 return err; /* checked elsewhere */
299
300 err = extract_number_checked(it, &value, &bytesUsed);
301 if (err)
302 return err;
303
304 bytesNeeded = 0;
305 if (value >= Value8Bit)
306 ++bytesNeeded;
307 if (value > 0xffU)
308 ++bytesNeeded;
309 if (value > 0xffffU)
310 bytesNeeded += 2;
311 if (value > 0xffffffffU)
312 bytesNeeded += 4;
313 if (bytesNeeded < bytesUsed)
314 return CborErrorOverlongEncoding;
315 return CborNoError;
316}

318static inline CborError validate_tag(CborValue *it, CborTag tag, uint32_t flags, int recursionLeft)
319{
320 CborType type = cbor_value_get_type(it);
321 const size_t knownTagCount = sizeof(knownTagData) / sizeof(knownTagData[0]);
322 const struct KnownTagData *tagData = knownTagData;
323 const struct KnownTagData * const knownTagDataEnd = knownTagData + knownTagCount;
324
325 if (!recursionLeft)
326 return CborErrorNestingTooDeep;
327 if (flags & CborValidateNoTags)
328 return CborErrorExcludedType;
329
330 /* find the tag data, if any */
331 for ( ; tagData != knownTagDataEnd; ++tagData) {
332 if (tagData->tag < tag)
333 continue;
334 if (tagData->tag > tag)
335 tagData = NULL;
336 break;
337 }
338 if (tagData == knownTagDataEnd)
339 tagData = NULL;
340
341 if (flags & CborValidateNoUnknownTags && !tagData) {
342 /* tag not found */
343 if (flags & CborValidateNoUnknownTagsSA && tag < 24)
344 return CborErrorUnknownTag;
345 if ((flags & CborValidateNoUnknownTagsSR) == CborValidateNoUnknownTagsSR && tag < 256)
346 return CborErrorUnknownTag;
347 if ((flags & CborValidateNoUnknownTags) == CborValidateNoUnknownTags)
348 return CborErrorUnknownTag;
349 }
350
351 if (flags & CborValidateTagUse && tagData && tagData->types) {
352 uint32_t allowedTypes = tagData->types;
353
354 /* correct Integer so it's not zero */
355 if (type == CborIntegerType)
356 type = (CborType)(type + 1);
357
358 while (allowedTypes) {
359 if ((uint8_t)(allowedTypes & 0xff) == type)
360 break;
361 allowedTypes >>= 8;
362 }
363 if (!allowedTypes)
364 return CborErrorInappropriateTagForType;
365 }
366
367 return validate_value(it, flags, recursionLeft);
368}

370#ifndef CBOR_NO_FLOATING_POINT
371static inline CborError validate_floating_point(CborValue *it, CborType type, uint32_t flags)
372{
373 CborError err;
374 int r;
375 double val;
376 float valf;
377 uint16_t valf16 = 0x7c01; /* dummy value, an infinity */
378
379 if (type != CborDoubleType) {
380 if (type == CborFloatType) {
381 err = cbor_value_get_float(it, &valf);
382 val = valf;
383 } else {
384# ifdef CBOR_NO_HALF_FLOAT_TYPE
385 (void)valf16;
386 return CborErrorUnsupportedType;
387# else
388 err = cbor_value_get_half_float(it, &valf16);
389 val = decode_half(valf16);
390# endif
391 }
392 } else {
393 err = cbor_value_get_double(it, &val);
394 }
395 cbor_assert(err == CborNoError); /* can't fail */
396
397 r = fpclassify(val);
398 if (r == FP_NAN || r == FP_INFINITE) {
399 if (flags & CborValidateFiniteFloatingPoint)
400 return CborErrorExcludedValue;
401 if (flags & CborValidateShortestFloatingPoint) {
402 if (type == CborDoubleType)
403 return CborErrorOverlongEncoding;
404# ifndef CBOR_NO_HALF_FLOAT_TYPE
405 if (type == CborFloatType)
406 return CborErrorOverlongEncoding;
407 if (r == FP_NAN && valf16 != 0x7e00)
408 return CborErrorImproperValue;
409 if (r == FP_INFINITE && valf16 != 0x7c00 && valf16 != 0xfc00)
410 return CborErrorImproperValue;
411# endif
412 }
413 }
414
415 if (flags & CborValidateShortestFloatingPoint && type > CborHalfFloatType) {
416 if (type == CborDoubleType) {
417 valf = (float)val;
418 if ((double)valf == val)
419 return CborErrorOverlongEncoding;
420 }
421# ifndef CBOR_NO_HALF_FLOAT_TYPE
422 if (type == CborFloatType) {
423 valf16 = encode_half(valf);
424 if (valf == decode_half(valf16))
425 return CborErrorOverlongEncoding;
426 }
427# endif
428 }
429
430 return CborNoError;
431}
432#endif

434static CborError validate_container(CborValue *it, int containerType, uint32_t flags, int recursionLeft)
435{
436 CborError err;
437 const uint8_t *previous = NULL;
438 const uint8_t *previous_end = NULL;
439
440 if (!recursionLeft)
441 return CborErrorNestingTooDeep;
442
443 while (!cbor_value_at_end(it)) {
444 const uint8_t *current = cbor_value_get_next_byte(it);
445
446 if (containerType == CborMapType) {
447 if (flags & CborValidateMapKeysAreString) {
448 CborType type = cbor_value_get_type(it);
449 if (type == CborTagType) {
450 /* skip the tags */
451 CborValue copy = *it;
452 err = cbor_value_skip_tag(&copy);
453 if (err)
454 return err;
455 type = cbor_value_get_type(&copy);
456 }
457 if (type != CborTextStringType)
458 return CborErrorMapKeyNotString;
459 }
460 }
461
462 err = validate_value(it, flags, recursionLeft);
463 if (err)
464 return err;
465
466 if (containerType != CborMapType)
467 continue;
468
469 if (flags & CborValidateMapIsSorted) {
470 if (it->parser->flags & CborParserFlag_ExternalSource)
471 return CborErrorUnimplementedValidation;
472 if (previous) {
473 size_t bytelen1 = (size_t)(previous_end - previous);
474 size_t bytelen2 = (size_t)(cbor_value_get_next_byte(it) - current);
475 int r = memcmp(previous, current, bytelen1 <= bytelen2 ? bytelen1 : bytelen2);
476
477 if (r == 0 && bytelen1 != bytelen2)
478 r = bytelen1 < bytelen2 ? -1 : +1;
479 if (r > 0)
480 return CborErrorMapNotSorted;
481 if (r == 0 && (flags & CborValidateMapKeysAreUnique) == CborValidateMapKeysAreUnique)
482 return CborErrorMapKeysNotUnique;
483 }
484
485 previous = current;
486 previous_end = cbor_value_get_next_byte(it);
487 }
488
489 /* map: that was the key, so get the value */
490 err = validate_value(it, flags, recursionLeft);
491 if (err)
492 return err;
493 }
494 return CborNoError;
495}

497static CborError validate_value(CborValue *it, uint32_t flags, int recursionLeft)
498{
499 CborError err;
500 CborType type = cbor_value_get_type(it);
501
502 if (cbor_value_is_length_known(it)) {
503 err = validate_number(it, type, flags);
504 if (err)
505 return err;
506 } else {
507 if (flags & CborValidateNoIndeterminateLength)
508 return CborErrorUnknownLength;
509 }
510
511 switch (type) {
512 case CborArrayType:
513 case CborMapType: {
514 /* recursive type */
515 CborValue recursed;
516 err = cbor_value_enter_container(it, &recursed);
517 if (!err)
518 err = validate_container(&recursed, type, flags, recursionLeft - 1);
519 if (err) {
520 copy_current_position(it, &recursed);
521 return err;
522 }
523 err = cbor_value_leave_container(it, &recursed);
524 if (err)
525 return err;
526 return CborNoError;
527 }
528
529 case CborIntegerType: {
530 uint64_t val;
531 err = cbor_value_get_raw_integer(it, &val);
532 cbor_assert(err == CborNoError); /* can't fail */
533
534 break;
535 }
536
537 case CborByteStringType:
538 case CborTextStringType: {
539 size_t n = 0;
540 const void *ptr;
541
542 err = cbor_value_begin_string_iteration(it);
543 if (err)
544 return err;
545
546 while (1) {
547 CborValue next;
548 err = _cbor_value_get_string_chunk(it, &ptr, &n, &next);
549 if (!err) {
550 err = validate_number(it, type, flags);
551 if (err)
552 return err;
553 }
554
555 *it = next;
556 if (err == CborErrorNoMoreStringChunks)
557 return cbor_value_finish_string_iteration(it);
558 if (err)
559 return err;
560
561 if (type == CborTextStringType && flags & CborValidateUtf8) {
562 err = validate_utf8_string(ptr, n);
563 if (err)
564 return err;
565 }
566 }
567
568 return CborNoError;
569 }
570
571 case CborTagType: {
572 CborTag tag;
573 err = cbor_value_get_tag(it, &tag);
574 cbor_assert(err == CborNoError); /* can't fail */
575
576 err = cbor_value_advance_fixed(it);
577 if (err)
578 return err;
579 err = validate_tag(it, tag, flags, recursionLeft - 1);
580 if (err)
581 return err;
582
583 return CborNoError;
584 }
585
586 case CborSimpleType: {
587 uint8_t simple_type;
588 err = cbor_value_get_simple_type(it, &simple_type);
589 cbor_assert(err == CborNoError); /* can't fail */
590 err = validate_simple_type(simple_type, flags);
591 if (err)
592 return err;
593 break;
594 }
595
596 case CborNullType:
597 case CborBooleanType:
598 break;
599
600 case CborUndefinedType:
601 if (flags & CborValidateNoUndefined)
602 return CborErrorExcludedType;
603 break;
604
605 case CborHalfFloatType:
606 case CborFloatType:
607 case CborDoubleType: {
608#ifdef CBOR_NO_FLOATING_POINT
609 return CborErrorUnsupportedType;
610#else
611 err = validate_floating_point(it, type, flags);
612 if (err)
613 return err;
614 break;
615#endif /* !CBOR_NO_FLOATING_POINT */
616 }
617
618 case CborInvalidType:
619 return CborErrorUnknownType;
620 }
621
622 err = cbor_value_advance_fixed(it);
623 return err;
624}

640CborError cbor_value_validate(const CborValue *it, uint32_t flags)
641{
642 CborValue value = *it;
643 CborError err = validate_value(&value, flags, CBOR_PARSER_MAX_RECURSIONS);
644 if (err)
645 return err;
646 if (flags & CborValidateCompleteData && can_read_bytes(&value, 1))
647 return CborErrorGarbageAtEnd;
648 return CborNoError;
649}
