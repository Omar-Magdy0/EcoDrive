98// get monotonic time in ns
99static inline uint64_t now_ns(void) {
100 struct timespec ts;
101 clock_gettime(CLOCK_MONOTONIC, &ts);
102 return (uint64_t)ts.tv_sec * 1000000000ULL + ts.tv_nsec;
103}

105// register a timer (frequency in Hz)
106bool register_timer(vtimer_manager_t* mgr, timer_callback_t cb, uint64_t timestep_ns) {
107 if (mgr->timer_index >= HOST_TIMERS) return false;
108
109 vtimer_t* t = &mgr->timers[mgr->timer_index++];
110 t->periodic_time_ns = (timestep_ns);
111 t->last_time_ns = now_ns();
112 t->cb = cb;
113
114 if(timestep_ns < mgr->min_timestep_ns)
115 {
116 mgr->min_timestep_ns = timestep_ns;
117 }
118
119 return true;
120}

122void* tick_thread(void* arg)
123{
124 const uint64_t sleep_ns = 1000000ULL; // 1 ms coarse sleep
125 uint64_t sim_time_ns = now_ns(); // simulation time anchored to wall clock
126 uint64_t last_wall_ns = sim_time_ns;
127
128 // determine smallest timestep needed by any timer
129 uint64_t dt_ns = timer_manager.min_timestep_ns; // example 10 us, can be min of all timers
130
131 while (1)
132 {
133 dt_ns = timer_manager.min_timestep_ns;
134 uint64_t current_wall_ns = now_ns();
135 uint64_t elapsed_ns = current_wall_ns - last_wall_ns;
136 last_wall_ns = current_wall_ns;
137
138 uint64_t steps = elapsed_ns / dt_ns;
139 for (uint64_t i = 0; i < steps; i++)
140 {
141 sim_time_ns += dt_ns;
142
143 // unified check of all timers
144 for (int t_idx = 0; t_idx < timer_manager.timer_index; t_idx++)
145 {
146 vtimer_t* t = &timer_manager.timers[t_idx];
147 if (!t->cb) continue;
148
149 if (sim_time_ns - t->last_time_ns >= t->periodic_time_ns)
150 {
151 t->last_time_ns += t->periodic_time_ns; // advance by one period
152 t->cb();
153 }
154 }
155 }
156
157 // coarse sleep to avoid spinning CPU
158 struct timespec req = {0, sleep_ns};
159 nanosleep(&req, NULL);
160 }
161}
