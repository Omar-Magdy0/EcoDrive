224CborError parseOne(CborValue *it, QString *parsed, int flags)
225{
226 char *buffer;
227 size_t size;
228
229 FILE *f = open_memstream(&buffer, &size);
230 CborError err = cbor_value_to_json_advance(f, it, flags);
231 fclose(f);
232
233 *parsed = QString::fromLatin1(buffer);
234 free(buffer);
235 return err;
236}

238bool compareFailed = true;
239void compareOne_real(const QByteArray &data, const QString &expected, int flags, int line)
240{
241 compareFailed = true;
242 CborParser parser;
243 CborValue first;
244 CborError err = cbor_parser_init(reinterpret_cast<const quint8 *>(data.constData()), data.length(), 0, &parser, &first);
245 QVERIFY2(!err, QByteArray::number(line) + ": Got error \"" + cbor_error_string(err) + "\"");
246
247 QString decoded;
248 err = parseOne(&first, &decoded, flags);
249 QVERIFY2(!err, QByteArray::number(line) + ": Got error \"" + cbor_error_string(err) +
250 "\"; decoded stream:\n" + decoded.toLatin1());
251 QCOMPARE(decoded, expected);
252
253 // check that we consumed everything
254 QCOMPARE((void*)cbor_value_get_next_byte(&first), (void*)data.constEnd());
255
256 compareFailed = false;
257}
258#define compareOne(data, expected, flags) \
259 compareOne_real(data, expected, flags, __LINE__); \
260 if (compareFailed) return

683void compareMetaData(QByteArray data, const QString &expected, int otherFlags = 0)
684{
685 QString decoded;
686
687 // needs to be in one map, with the entry called "v"
688 data = "\xa1\x61v" + data;
689
690 {
691 CborParser parser;
692 CborValue first;
693 CborError err = cbor_parser_init(reinterpret_cast<const quint8 *>(data.constData()), data.length(), 0, &parser, &first);
694 QVERIFY2(!err, QByteArrayLiteral(": Got error \"") + cbor_error_string(err) + "\"");
695
696 err = parseOne(&first, &decoded, CborConvertAddMetadata | otherFlags);
697 QVERIFY2(!err, QByteArrayLiteral(": Got error \"") + cbor_error_string(err) +
698 "\"; decoded stream:\n" + decoded.toLatin1());
699
700 // check that we consumed everything
701 QCOMPARE((void*)cbor_value_get_next_byte(&first), (void*)data.constEnd());
702 }
703
704 QVERIFY(decoded.startsWith("{\"v\":"));
705 QVERIFY(decoded.endsWith('}'));
706// qDebug() << "was" << decoded;
707
708 // extract just the metadata
709 static const char needle[] = "\"v$cbor\":{";
710 int pos = decoded.indexOf(needle);
711 QCOMPARE(pos == -1, expected.isEmpty());
712 if (pos != -1) {
713 decoded.chop(2);
714 decoded = std::move(decoded).mid(pos + strlen(needle));
715 QCOMPARE(decoded, expected);
716 }
717}
