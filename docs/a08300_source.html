86static inline void elcore_rstream_commitWrite(elcore_rstream_t* cb, uint16_t count) {
87 uint16_t occupied = elcore_rstream_occupied(cb);
88 cb->stats.bytes_written += count * cb->item_size;
89 if (occupied > cb->stats.high_watermark) {
90 cb->stats.high_watermark = occupied;

104static inline bool elcore_rstream_reserveWrite(elcore_rstream_t* cb,
105 uint16_t count,
106 void** writeptr1,
107 uint16_t* cont1,
108 void** writeptr2,
109 uint16_t* cont2)
110{
111 if (count > elcore_rstream_freeSpace(cb))
112 {
113 cb->stats.failed_writes++;
114 return false;
115 }
116
117 uint16_t cont_space;
118
119 if (cb->head >= cb->tail)
120 cont_space = cb->capacity - cb->head - (cb->tail == 0 ? 1 : 0);
121 else
122 cont_space = cb->tail - cb->head - 1;
123
124 *writeptr1 = cb->buffer + cb->head * cb->item_size;
125
126 if (cont_space >= count) {
127 *writeptr2 = NULL;
128 *cont1 = count;
129 *cont2 = 0;
130 } else {
131 *writeptr2 = cb->buffer;
132 *cont1 = cont_space;
133 *cont2 = count - cont_space;
134 }
135
136 return true;
