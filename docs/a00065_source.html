216static inline void put16(void *where, uint16_t v)
217{

222/* Note: Since this is currently only used in situations where OOM is the only
223 * valid error, we KNOW this to be true. Thus, this function now returns just 'true',
224 * but if in the future, any function starts returning a non-OOM error, this will need
225 * to be changed to the test. At the moment, this is done to prevent more branches
226 * being created in the tinycbor output */
227static inline bool isOomError(CborError err)
228{
229 if (CBOR_ENCODER_WRITER_CONTROL < 0)
230 return true;
231
232 /* CborErrorOutOfMemory is the only negative error code, intentionally
233 * so we can write the test like this */
234 return (int)err < 0;
235}

249static inline bool would_overflow(CborEncoder *encoder, size_t len)
250{
251 ptrdiff_t remaining = (ptrdiff_t)encoder->end;
252 remaining -= remaining ? (ptrdiff_t)encoder->data.ptr : encoder->data.bytes_needed;
253 remaining -= (ptrdiff_t)len;
254 return unlikely(remaining < 0);
255}

257static inline void advance_ptr(CborEncoder *encoder, size_t n)
258{

265static inline CborError append_to_buffer(CborEncoder *encoder, const void *data, size_t len,
266 CborEncoderAppendType appendType)
267{
268 if (CBOR_ENCODER_WRITER_CONTROL >= 0) {
269 if (encoder->flags & CborIteratorFlag_WriterFunction || CBOR_ENCODER_WRITER_CONTROL != 0) {
270# ifdef CBOR_ENCODER_WRITE_FUNCTION
271 return CBOR_ENCODER_WRITE_FUNCTION(encoder->end, data, len, appendType);
272# else
273 return encoder->data.writer(encoder->end, data, len, appendType);
274# endif
275 }
276 }
277
278#if CBOR_ENCODER_WRITER_CONTROL <= 0
279 if (would_overflow(encoder, len)) {
280 if (encoder->end != NULL) {
281 len -= encoder->end - encoder->data.ptr;
282 encoder->end = NULL;
283 encoder->data.bytes_needed = 0;
284 }
285
286 advance_ptr(encoder, len);
287 return CborErrorOutOfMemory;
288 }
289
290 memcpy(encoder->data.ptr, data, len);
291 encoder->data.ptr += len;
292#endif
293 return CborNoError;
294}

301static inline CborError encode_number_no_update(CborEncoder *encoder, uint64_t ui, uint8_t shiftedMajorType)
302{
303 /* Little-endian would have been so much more convenient here:
304 * We could just write at the beginning of buf but append_to_buffer
305 * only the necessary bytes.
306 * Since it has to be big endian, do it the other way around:
307 * write from the end. */
308 uint64_t buf[2];
309 uint8_t *const bufend = (uint8_t *)buf + sizeof(buf);
310 uint8_t *bufstart = bufend - 1;
311 put64(buf + 1, ui); /* we probably have a bunch of zeros in the beginning */
312
313 if (ui < Value8Bit) {
314 *bufstart += shiftedMajorType;
315 } else {
316 uint8_t more = 0;
317 if (ui > 0xffU)
318 ++more;
319 if (ui > 0xffffU)
320 ++more;
321 if (ui > 0xffffffffU)
322 ++more;
323 bufstart -= (size_t)1 << more;
324 *bufstart = shiftedMajorType + Value8Bit + more;
325 }
326
327 return append_to_buffer(encoder, bufstart, bufend - bufstart, CborEncoderAppendCborData);
328}

372CborError cbor_encode_int(CborEncoder *encoder, int64_t value)
373{
374 /* adapted from code in RFC 7049 appendix C (pseudocode) */
375 uint64_t ui = value >> 63; /* extend sign to whole length */
376 uint8_t majorType = ui & 0x20; /* extract major type */
377 ui ^= value; /* complement negatives */
378 return encode_number(encoder, ui, majorType);
379}

409CborError cbor_encode_floating_point(CborEncoder *encoder, CborType fpType, const void *value)
410{
411 unsigned size;
412 uint8_t buf[1 + sizeof(uint64_t)];
413 cbor_assert(fpType == CborHalfFloatType || fpType == CborFloatType || fpType == CborDoubleType);
414 buf[0] = fpType;
415
416 size = 2U << (fpType - CborHalfFloatType);
417 if (size == 8)
418 put64(buf + 1, *(const uint64_t*)value);
419 else if (size == 4)
420 put32(buf + 1, *(const uint32_t*)value);
421 else
422 put16(buf + 1, *(const uint16_t*)value);
423 saturated_decrement(encoder);
424 return append_to_buffer(encoder, buf, size + 1, CborEncoderAppendCborData);
425}

497static CborError create_container(CborEncoder *encoder, CborEncoder *container, size_t length, uint8_t shiftedMajorType)
498{
499 CborError err;
500 container->data.ptr = encoder->data.ptr;
501 container->end = encoder->end;
502 saturated_decrement(encoder);
503 container->remaining = length + 1; /* overflow ok on CborIndefiniteLength */
504
505 cbor_static_assert((int)CborIteratorFlag_ContainerIsMap_ == (int)CborIteratorFlag_ContainerIsMap);
506 cbor_static_assert(((MapType << MajorTypeShift) & CborIteratorFlag_ContainerIsMap) == CborIteratorFlag_ContainerIsMap);
507 cbor_static_assert(((ArrayType << MajorTypeShift) & CborIteratorFlag_ContainerIsMap) == 0);
508 container->flags = shiftedMajorType & CborIteratorFlag_ContainerIsMap;
509 if (CBOR_ENCODER_WRITER_CONTROL == 0)
510 container->flags |= encoder->flags & CborIteratorFlag_WriterFunction;
511
512 if (length == CborIndefiniteLength) {
513 container->flags |= CborIteratorFlag_UnknownLength;
514 err = append_byte_to_buffer(container, shiftedMajorType + IndefiniteLength);
515 } else {
516 if (shiftedMajorType & CborIteratorFlag_ContainerIsMap)
517 container->remaining += length;
518 err = encode_number_no_update(container, length, shiftedMajorType);
519 }
520 return err;
521}

580CborError cbor_encoder_close_container(CborEncoder *parentEncoder, const CborEncoder *containerEncoder)
581{
582 // synchronise buffer state with that of the container
583 parentEncoder->end = containerEncoder->end;
584 parentEncoder->data = containerEncoder->data;
585
586 if (containerEncoder->flags & CborIteratorFlag_UnknownLength)
587 return append_byte_to_buffer(parentEncoder, BreakByte);
588
589 if (containerEncoder->remaining != 1)
590 return containerEncoder->remaining == 0 ? CborErrorTooManyItems : CborErrorTooFewItems;
591
592 if (!parentEncoder->end)
593 return CborErrorOutOfMemory; /* keep the state */
594
595 return CborNoError;
596}
