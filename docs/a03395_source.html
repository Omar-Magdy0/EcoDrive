41void mcobs_encode(uint8_t* buf, uint8_t len){
42 if (len == 0) {
43 buf[0] = 0x01; // Special case for empty data
44 return;
45 }
46
47 uint8_t read_pos = 1;
48 uint8_t code_pos = 0; // Position to write next code byte
49 uint8_t distance = 1; // Distance to next zero
50
51 while (read_pos < len) {
52 if (buf[read_pos] == 0) {
53 // Found zero: write distance at code_pos
54 buf[code_pos] = distance;
55 // Start new block
56 code_pos = read_pos;
57 distance = 1;
58 } else {
59 distance++;
60 }
61 read_pos++;
62 }
63 // Write last distance
64 buf[code_pos] = distance;
65}

77 // Start new block
78 } else {

97uint8_t aebf_encode_frame(uint8_t* output_buffer,
98 uint8_t device_id, uint16_t service_id,
99 uint8_t payload_len)
100{
101 bool has_crc = false;
102 if (payload_len > AEBF_MAX_PAYLOAD_SIZE) return false;
103 if (device_id > AEBF_MAX_DEVICES || service_id > AEBF_MAX_SERVICES) return false;
104
105 output_buffer[0] = 0x00;
106 output_buffer[1] = payload_len;
107
108 //SHOULD BE PROPER CRC
109 output_buffer[3] = (device_id << 2) | (service_id >> 8);
110 output_buffer[4] = service_id & 0xFF;
111 if(service_id >= 0x00 && service_id <= 0xC0) has_crc = true;
112 uint16_t crc16_modbus;
113 if(has_crc){
114 crc16_modbus = aebf_crc16_modbus(output_buffer + 3, payload_len + 2);
115 }else{
116 crc16_modbus = 0xFFFF;
117 }
118 output_buffer[AEBF_FRAMED_SIZE(payload_len) - 2] = crc16_modbus >> 8;
119 output_buffer[AEBF_FRAMED_SIZE(payload_len) - 1] = crc16_modbus & 0xFF;
120 mcobs_encode(output_buffer+2, AEBF_FRAMED_SIZE(payload_len) - 2);
121 return AEBF_ERROR_FRAME_OK;
122}

126uint8_t aebf_decode_frame(uint8_t* frame_buffer, uint16_t frame_len,
127 uint8_t* device_id, uint16_t* service_id,
128 uint8_t* payload_len)
129{
130 bool has_crc = false;
131 if(!aebf_is_frame_cplt(frame_buffer, frame_len)){
132 return AEBF_ERROR_FRAME_LENGTH_MISMATCH;
133 }else
134 {
135 *payload_len = frame_buffer[1];
136 mcobs_decode(frame_buffer+2, AEBF_FRAMED_SIZE(*payload_len) - 2);
137
138 *device_id = (frame_buffer[3] >> 2) & 0x3F;
139 *service_id = ((frame_buffer[3] & 0x03) << 8) | frame_buffer[4];
140 if(*service_id >= 0x00 && *service_id <= 0xC0) has_crc = true;
141 if(has_crc){
142 uint16_t crc16_modbus_calc = aebf_crc16_modbus(frame_buffer + 3, *payload_len + 2);
143 uint16_t crc16_modbus_recv = (frame_buffer[AEBF_FRAMED_SIZE(*payload_len) - 2] << 8) | frame_buffer[AEBF_FRAMED_SIZE(*payload_len) - 1];
144 if (crc16_modbus_calc != crc16_modbus_recv) return AEBF_ERROR_CRC_MISMATCH;
145 }
146 return AEBF_ERROR_FRAME_OK;
147 }

185void aebfStream_process(aebfStream_t* stream, const uint8_t* data, uint16_t length) {
186 if (!stream || !data || length == 0) {
187 return;
188 }
189
190 // Process each incoming byte
191 for (uint16_t i = 0; i < length; i++) {
192 uint8_t byte = data[i];
193
194 switch (stream->state) {
195
196 case STATE_HUNT_SYNC:
197 // Looking for sync byte (0x00)
198 if (byte == AEBF_SYNC_BYTE) {
199 // Found sync - start accumulating
200 stream->frame_buf[0] = byte;
201 stream->frame_pos = 1;
202 stream->state = STATE_READ_LENGTH;
203 stream->frames_received++;
204 } else {
205 // Not sync, skip (don't accumulate)
206 }
207 break;
208
209 // Got sync, next byte is payload length
210 case STATE_READ_LENGTH: {
211 stream->frame_buf[stream->frame_pos++] = byte;
212 stream->frame_length = byte;
213
214 // Expected Remaining Bytes: Cobs Code (1) + device/service(2) + payload(N) + crc(2)
215 stream->bytes_expected = AEBF_FRAMED_SIZE(stream->frame_length) - 2;
216
217 // Sanity check payload length
218 if (stream->bytes_expected > AEBF_FRAMED_SIZE(AEBF_MAX_PAYLOAD_SIZE) - 2 || stream->bytes_expected <= (AEBF_FRAMED_SIZE(0) - 2) ) {
219 // Invalid length, resync
220 if (stream->on_error) {
221 stream->on_error(AEBF_ERROR_PAYLOAD_SIZE, stream->user_data);
222 }
223 stream->frames_corrupted++;
224 stream->frame_pos = 0;
225 stream->state = STATE_HUNT_SYNC;
226 } else {
227 stream->state = STATE_READ_RESTFRAME;
228 }
229 break;
230 }
231
232 case STATE_READ_RESTFRAME: {
233 stream->frame_buf[stream->frame_pos++] = byte;
234
235 if(byte == AEBF_SYNC_BYTE) {
236 // Found sync byte in payload - resync
237 if (stream->on_error) {
238 stream->on_error(AEBF_ERROR_FRAME_LENGTH_MISMATCH, stream->user_data);
239 }
240 stream->frames_corrupted++;
241 stream->frame_pos = 0;
242 stream->state = STATE_HUNT_SYNC;
243 break;
244 }
245
246 stream->bytes_expected--;
247 if (stream->bytes_expected == 0) {
248 // Frame complete
249 stream->state = STATE_VALIDATE;
250 }else{
251 break;
252 }
253
254 }
255
256
257 case STATE_VALIDATE: {
258 // Frame is complete in frame_buf
259 uint16_t frame_len = stream->frame_pos;
260 uint8_t deviceID;
261 uint16_t serviceID;
262 uint8_t payload_len;
263
264 uint8_t error_code = aebf_decode_frame(stream->frame_buf, frame_len, &deviceID, &serviceID, &payload_len);
265 // Call user callback with complete frame
266 if (error_code != AEBF_ERROR_FRAME_OK)
267 {
268 if (stream->on_error) {
269 stream->on_error(error_code, stream->user_data);
270 }
271 stream->frames_corrupted++;
272 } else
273 {
274 if (stream->on_frame)
275 {
276 stream->on_frame(deviceID, serviceID, stream->frame_buf + 5, payload_len, stream->user_data);
277 }
278 stream->frames_valid++;
279 }
280
281 stream->frame_pos = 0;
282 stream->bytes_expected = 0;
283 stream->state = STATE_HUNT_SYNC;
284 break;
285 }
286
287 default:
288 stream->state = STATE_HUNT_SYNC;
289 break;
290 }
291 }
