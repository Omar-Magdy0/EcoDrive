144static inline bool convertToUint64(double v, uint64_t *absolute)
145{
146 double supremum;
147 v = fabs(v);
148
149 /* C11 standard section 6.3.1.4 "Real floating and integer" says:
150 *
151 * 1 When a finite value of real floating type is converted to an integer
152 * type other than _Bool, the fractional part is discarded (i.e., the
153 * value is truncated toward zero). If the value of the integral part
154 * cannot be represented by the integer type, the behavior is undefined.
155 *
156 * So we must perform a range check that v <= UINT64_MAX, but we can't use
157 * UINT64_MAX + 1.0 because the standard continues:
158 *
159 * 2 When a value of integer type is converted to a real floating type, if
160 * the value being converted can be represented exactly in the new type,
161 * it is unchanged. If the value being converted is in the range of
162 * values that can be represented but cannot be represented exactly, the
163 * result is either the nearest higher or nearest lower representable
164 * value, chosen in an implementation-defined manner.
165 */
166 supremum = -2.0 * INT64_MIN; /* -2 * (- 2^63) == 2^64 */
167 if (!(v < supremum)) /* out of range or NaN */
168 return false;
169
170 /* Now we can convert, these two conversions cannot be UB */
171 *absolute = v;
172 return *absolute == v;
173}
174#endif

181static CborError hexDump(CborStreamFunction stream, void *out, const void *ptr, size_t n)
182{
183 const uint8_t *buffer = (const uint8_t *)ptr;
184 CborError err = CborNoError;
185 while (n-- && !err)
186 err = stream(out, "%02" PRIx8, *buffer++);
187
188 return err;
189}

191/* This function decodes buffer as UTF-8 and prints as escaped UTF-16.
192 * On UTF-8 decoding error, it returns CborErrorInvalidUtf8TextString */
193static CborError utf8EscapedDump(CborStreamFunction stream, void *out, const void *ptr, size_t n)
194{
195 const uint8_t *buffer = (const uint8_t *)ptr;
196 const uint8_t * const end = buffer + n;
197 CborError err = CborNoError;
198
199 while (buffer < end && !err) {
200 uint32_t uc = get_utf8(&buffer, end);
201 if (uc == ~0U)
202 return CborErrorInvalidUtf8TextString;
203
204 if (uc < 0x80) {
205 /* single-byte UTF-8 */
206 unsigned char escaped = (unsigned char)uc;
207 if (uc < 0x7f && uc >= 0x20 && uc != '\\' && uc != '"') {
208 err = stream(out, "%c", (char)uc);
209 continue;
210 }
211
212 /* print as an escape sequence */
213 switch (uc) {
214 case '"':
215 case '\\':
216 break;
217 case '\b':
218 escaped = 'b';
219 break;
220 case '\f':
221 escaped = 'f';
222 break;
223 case '\n':
224 escaped = 'n';
225 break;
226 case '\r':
227 escaped = 'r';
228 break;
229 case '\t':
230 escaped = 't';
231 break;
232 default:
233 goto print_utf16;
234 }
235 err = stream(out, "\\%c", escaped);
236 continue;
237 }
238
239 /* now print the sequence */
240 if (uc > 0xffffU) {
241 /* needs surrogate pairs */
242 err = stream(out, "\\u%04" PRIX32 "\\u%04" PRIX32,
243 (uc >> 10) + 0xd7c0, /* high surrogate */
244 (uc % 0x0400) + 0xdc00);
245 } else {
246print_utf16:
247 /* no surrogate pair needed */
248 err = stream(out, "\\u%04" PRIX32, uc);
249 }
250 }
251 return err;
252}

254static const char *resolve_indicator(const CborValue *it, int flags)
255{
256 static const char indicators[8][3] = {
257 "_0", "_1", "_2", "_3",
258 "", "", "", /* these are not possible */
259 "_"
260 };
261 const char *no_indicator = indicators[5]; /* empty string */
262 uint8_t additional_information;
263 uint8_t expected_information;
264 uint64_t value;
265 CborError err;
266
267 if (!read_bytes(it, &additional_information, 0, 1))
268 return NULL; /* CborErrorUnexpectedEOF */
269
270 additional_information &= SmallValueMask;
271 if (additional_information < Value8Bit)
272 return no_indicator;
273
274 /* determine whether to show anything */
275 if ((flags & CborPrettyIndicateIndeterminateLength) &&
276 additional_information == IndefiniteLength)
277 return indicators[IndefiniteLength - Value8Bit];
278 if ((flags & CborPrettyIndicateOverlongNumbers) == 0)
279 return no_indicator;
280
281 err = extract_number_checked(it, &value, NULL);
282 if (err)
283 return NULL; /* CborErrorUnexpectedEOF */
284
285 expected_information = Value8Bit - 1;
286 if (value >= Value8Bit)
287 ++expected_information;
288 if (value > 0xffU)
289 ++expected_information;
290 if (value > 0xffffU)
291 ++expected_information;
292 if (value > 0xffffffffU)
293 ++expected_information;
294 return expected_information == additional_information ?
295 no_indicator :
296 indicators[additional_information - Value8Bit];
297}

305static CborError container_to_pretty(CborStreamFunction stream, void *out, CborValue *it, CborType containerType,
306 int flags, int recursionsLeft)
307{
308 const char *comma = "";
309 CborError err = CborNoError;
310
311 if (recursionsLeft <= 0) {
312 printRecursionLimit(stream, out);
313 while (!cbor_value_at_end(it) && !err) {
314 err = cbor_value_advance(it);
315 }
316 return err; /* do allow the dumping to continue (if the advance was OK) */
317 }
318
319 while (!cbor_value_at_end(it) && !err) {
320 err = stream(out, "%s", comma);
321 comma = ", ";
322
323 if (!err)
324 err = value_to_pretty(stream, out, it, flags, recursionsLeft);
325
326 if (containerType == CborArrayType)
327 continue;
328
329 /* map: that was the key, so get the value */
330 if (!err)
331 err = stream(out, ": ");
332 if (!err)
333 err = value_to_pretty(stream, out, it, flags, recursionsLeft);
334 }
335 return err;
336}

338static CborError value_to_pretty(CborStreamFunction stream, void *out, CborValue *it, int flags, int recursionsLeft)
339{
340 CborError err = CborNoError;
341 CborType type = cbor_value_get_type(it);
342 switch (type) {
343 case CborArrayType:
344 case CborMapType: {
345 /* recursive type */
346 CborValue recursed;
347 const char *indicator = get_indicator(it, flags);
348 const char *space = *indicator ? " " : indicator;
349
350 err = stream(out, "%c%s%s", type == CborArrayType ? '[' : '{', indicator, space);
351 if (err)
352 return err;
353
354 err = cbor_value_enter_container(it, &recursed);
355 if (err) {
356 copy_current_position(it, &recursed);
357 return err; /* parse error */
358 }
359 /* N.B. recursionsLeft can be zero, in which case container_to_pretty is called with
360 * recursionsLeft = -1 and reports that nesting is too deep.
361 */
362 err = container_to_pretty(stream, out, &recursed, type, flags, recursionsLeft - 1);
363 if (err) {
364 copy_current_position(it, &recursed);
365 return err; /* parse error */
366 }
367 err = cbor_value_leave_container(it, &recursed);
368 if (err)
369 return err; /* parse error */
370
371 return stream(out, type == CborArrayType ? "]" : "}");
372 }
373
374 case CborIntegerType: {
375 uint64_t val;
376 cbor_value_get_raw_integer(it, &val); /* can't fail */
377
378 if (cbor_value_is_unsigned_integer(it)) {
379 err = stream(out, "%" PRIu64, val);
380 } else {
381 /* CBOR stores the negative number X as -1 - X
382 * (that is, -1 is stored as 0, -2 as 1 and so forth) */
383 if (++val) { /* unsigned overflow may happen */
384 err = stream(out, "-%" PRIu64, val);
385 } else {
386 /* overflown
387 * 0xffff`ffff`ffff`ffff + 1 =
388 * 0x1`0000`0000`0000`0000 = 18446744073709551616 (2^64) */
389 err = stream(out, "-18446744073709551616");
390 }
391 }
392 if (!err)
393 err = stream(out, "%s", get_indicator(it, flags));
394 break;
395 }
396
397 case CborByteStringType:
398 case CborTextStringType: {
399 size_t n = 0;
400 const void *ptr;
401 bool showingFragments = (flags & CborPrettyShowStringFragments) && !cbor_value_is_length_known(it);
402 const char *separator = "";
403 char close = '\'';
404 char open[3] = "h'";
405 const char *indicator = NULL;
406
407 if (type == CborTextStringType) {
408 close = open[0] = '"';
409 open[1] = '\0';
410 }
411
412 if (showingFragments)
413 err = stream(out, "(_ ");
414 else
415 err = stream(out, "%s", open);
416
417 if (!err)
418 err = cbor_value_begin_string_iteration(it);
419 while (!err) {
420 if (showingFragments || indicator == NULL) {
421 /* any iteration, except the second for a non-chunked string */
422 indicator = resolve_indicator(it, flags);
423 }
424
425 err = _cbor_value_get_string_chunk(it, &ptr, &n, it);
426 if (err == CborErrorNoMoreStringChunks) {
427 err = cbor_value_finish_string_iteration(it);
428 break;
429 }
430
431 if (!err && showingFragments)
432 err = stream(out, "%s%s", separator, open);
433 if (!err)
434 err = (type == CborByteStringType ?
435 hexDump(stream, out, ptr, n) :
436 utf8EscapedDump(stream, out, ptr, n));
437 if (!err && showingFragments) {
438 err = stream(out, "%c%s", close, indicator);
439 separator = ", ";
440 }
441 }
442
443 if (!err) {
444 if (showingFragments)
445 err = stream(out, ")");
446 else
447 err = stream(out, "%c%s", close, indicator);
448 }
449 return err;
450 }
451
452 case CborTagType: {
453 CborTag tag;
454 cbor_value_get_tag(it, &tag); /* can't fail */
455 err = stream(out, "%" PRIu64 "%s(", tag, get_indicator(it, flags));
456 if (!err)
457 err = cbor_value_advance_fixed(it);
458 if (!err && recursionsLeft > 0)
459 err = value_to_pretty(stream, out, it, flags, recursionsLeft - 1);
460 else if (!err)
461 printRecursionLimit(stream, out);
462 if (!err)
463 err = stream(out, ")");
464 return err;
465 }
466
467 case CborSimpleType: {
468 /* simple types can't fail and can't have overlong encoding */
469 uint8_t simple_type;
470 cbor_value_get_simple_type(it, &simple_type);
471 err = stream(out, "simple(%" PRIu8 ")", simple_type);
472 break;
473 }
474
475 case CborNullType:
476 err = stream(out, "null");
477 break;
478
479 case CborUndefinedType:
480 err = stream(out, "undefined");
481 break;
482
483 case CborBooleanType: {
484 bool val;
485 cbor_value_get_boolean(it, &val); /* can't fail */
486 err = stream(out, val ? "true" : "false");
487 break;
488 }
489
490#ifndef CBOR_NO_FLOATING_POINT
491 case CborDoubleType: {
492 const char *suffix;
493 double val;
494 int r;
495 uint64_t ival;
496
497 if (false) {
498 float f;
499 case CborFloatType:
500 cbor_value_get_float(it, &f);
501 val = f;
502 suffix = flags & CborPrettyNumericEncodingIndicators ? "_2" : "f";
503 } else if (false) {
504 uint16_t f16;
505 case CborHalfFloatType:
506#ifndef CBOR_NO_HALF_FLOAT_TYPE
507 cbor_value_get_half_float(it, &f16);
508 val = decode_half(f16);
509 suffix = flags & CborPrettyNumericEncodingIndicators ? "_1" : "f16";
510#else
511 (void)f16;
512 err = CborErrorUnsupportedType;
513 break;
514#endif
515 } else {
516 cbor_value_get_double(it, &val);
517 suffix = "";
518 }
519
520 if ((flags & CborPrettyNumericEncodingIndicators) == 0) {
521 r = fpclassify(val);
522 if (r == FP_NAN || r == FP_INFINITE)
523 suffix = "";
524 }
525
526 if (convertToUint64(val, &ival)) {
527 /* this double value fits in a 64-bit integer, so show it as such
528 * (followed by a floating point suffix, to disambiguate) */
529 err = stream(out, "%s%" PRIu64 ".%s", val < 0 ? "-" : "", ival, suffix);
530 } else {
531 /* this number is definitely not a 64-bit integer */
532 err = stream(out, "%." DBL_DECIMAL_DIG_STR "g%s", val, suffix);
533 }
534 break;
535 }
536#else
537 case CborDoubleType:
538 case CborFloatType:
539 case CborHalfFloatType:
540 err = CborErrorUnsupportedType;
541 break;
542#endif /* !CBOR_NO_FLOATING_POINT */
543
544 case CborInvalidType:
545 err = stream(out, "invalid");
546 if (err)
547 return err;
548 return CborErrorUnknownType;
549 }
550
551 if (!err)
552 err = cbor_value_advance_fixed(it);
553 return err;
554}
