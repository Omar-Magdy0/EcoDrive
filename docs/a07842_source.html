83uint16_t adc1_Sample_Single_Channel_Temporary(uint32_t channel)
84{
85 // Save current configuration
86 uint32_t old_sqr1 = ADC1->SQR1;
87 uint32_t old_sqr2 = ADC1->SQR2;
88 uint32_t old_sqr3 = ADC1->SQR3;
89
90 // Configure for single conversion
91 // Configure for single conversion USING LL FUNCTION
92 LL_ADC_REG_SetSequencerLength(ADC1, LL_ADC_REG_SEQ_SCAN_DISABLE); // 1 conversion
93 LL_ADC_REG_SetSequencerRanks(ADC1, LL_ADC_REG_RANK_1, channel); // Handles conversion
94
95 // Start conversion
96 LL_ADC_REG_StartConversionSWStart(ADC1);
97 while (!LL_ADC_IsActiveFlag_EOCS(ADC1));
98 uint16_t result = LL_ADC_REG_ReadConversionData12(ADC1);
99
100 // Restore original configuration
101 ADC1->SQR1 = old_sqr1;
102 ADC1->SQR2 = old_sqr2;

114void mc3p_adc_calibrate(eldriver_mc3p_t *h)
115{
116 //Enable VREFINT and internalTEMP sensor
117 ADC->CCR |= ADC_CCR_TSVREFE_Msk;
118 #ifndef ELDRIVER_MC3P_VREFEXT
119
120 // Wait for internal references to stabilize (IMPORTANT!)
121 volatile uint32_t wait = 10000;
122 while(wait){wait = wait - 1;};
123
124 // Set longer sampling time for VREFINT
125 LL_ADC_SetChannelSamplingTime(ADC1, LL_ADC_CHANNEL_VREFINT, LL_ADC_SAMPLINGTIME_480CYCLES);
126
127 // Sample VREFINT multiple times for accuracy
128 uint32_t sum = 0;
129 const uint8_t num_samples = 30;
130
131 //Discard first sample
132
133 adc1_Sample_Single_Channel_Temporary(LL_ADC_CHANNEL_VREFINT);
134 for (int i = 0; i < num_samples; i++) {
135 sum += adc1_Sample_Single_Channel_Temporary(LL_ADC_CHANNEL_VREFINT);
136 }
137 uint32_t inRef_div_adcRef = sum / num_samples;
138
139 h->internal_ref_V = (*VREFIN_CAL_ADDR / (float)((1<<ELDRIVER_MC3P_ADCRES) - 1) ) * VREFIN_CAL_VOLTAGE;
140
141 // Calculate actual ADC reference voltage (your supply voltage)
142 (h->adc_ref_V) = h->internal_ref_V * ((1<<ELDRIVER_MC3P_ADCRES) - 1) / (float)inRef_div_adcRef;
143 #else
144
145 adc_ref_V = externalRefVoltage_V;
146 #endif
147
148 (h->adc_to_uV) = (float)(h->adc_ref_V * 1000000 ) / (float)((1<<ELDRIVER_MC3P_ADCRES) - 1);
149}

295

298 LL_DMA_SetStreamPriorityLevel(MC3P_DMA_INSTANCE, MC3P_DMA_BACKGROUND_STREAM, LL_DMA_PRIORITY_MEDIUM);
299 LL_DMA_SetMode(MC3P_DMA_INSTANCE, MC3P_DMA_BACKGROUND_STREAM, LL_DMA_MODE_CIRCULAR);
300 LL_DMA_SetPeriphIncMode(MC3P_DMA_INSTANCE, MC3P_DMA_BACKGROUND_STREAM, LL_DMA_PERIPH_NOINCREMENT);
301 LL_DMA_SetMemoryIncMode(MC3P_DMA_INSTANCE, MC3P_DMA_BACKGROUND_STREAM, LL_DMA_MEMORY_INCREMENT);
302 LL_DMA_SetPeriphSize(MC3P_DMA_INSTANCE, MC3P_DMA_BACKGROUND_STREAM, LL_DMA_PDATAALIGN_HALFWORD);
303 LL_DMA_SetMemorySize(MC3P_DMA_INSTANCE, MC3P_DMA_BACKGROUND_STREAM, LL_DMA_MDATAALIGN_HALFWORD);
304 LL_DMA_DisableFifoMode(MC3P_DMA_INSTANCE, MC3P_DMA_BACKGROUND_STREAM);
305
306 // Set peripheral address (ADC1 data register)
307 LL_DMA_SetPeriphAddress(MC3P_DMA_INSTANCE, MC3P_DMA_BACKGROUND_STREAM, (uint32_t)&(ADC1->DR));
308
309 // Set memory address (double buffer start)
310 LL_DMA_SetMemoryAddress(MC3P_DMA_INSTANCE, MC3P_DMA_BACKGROUND_STREAM, (uint32_t)mc3p_bg_data[0]);
311
312 // Set data length (total samples across both buffers)

314
315 // Enable DMA transfer complete and half transfer interrupts
316 LL_DMA_EnableIT_TC(MC3P_DMA_INSTANCE, MC3P_DMA_BACKGROUND_STREAM);
317 LL_DMA_EnableIT_HT(MC3P_DMA_INSTANCE, MC3P_DMA_BACKGROUND_STREAM);

522void eldriver_mc3p_write_svm(eldriver_mc3p_t *h, int16_t alpha_q15, int16_t beta_q15)
523{
524 int32_t vmax, vmin, voff;
525 if(!IS_SVM_SECTOR(h->sector_last))
526 {
527 eldriver_mc3p_write_phase_state(h, ELDRIVER_MC3P_PHASE_COMP, ELDRIVER_MC3P_PHASE_COMP, ELDRIVER_MC3P_PHASE_COMP);
528 h->mode = ELDRIVER_MC3P_MODE_SVM;
529 }
530
531 /* αβ → phase (normalized Q15) */
532 h->dutyu_q15 = alpha_q15;
533
534 h->dutyv_q15 = (-(int32_t)Q15_HALF * alpha_q15
535 + (int32_t)Q15_SQRT3_BY_2 * beta_q15) >> 15;
536
537 h->dutyw_q15 = (-(int32_t)Q15_HALF * alpha_q15
538 - (int32_t)Q15_SQRT3_BY_2 * beta_q15) >> 15;
539
540 /* SVPWM zero-sequence injection */
541 uint8_t b0 = (h->dutyu_q15 >= 0);
542 uint8_t b1 = (h->dutyv_q15 >= 0);
543 uint8_t b2 = (h->dutyw_q15 >= 0);
544
545 // 3-bit code
546 uint8_t code = (b2 << 2) | (b1 << 1) | b0;
547 eldriver_mc3p_sector_t sector;
548 switch(code)
549 {
550 case 0b001: sector = 1; break;
551 case 0b011: sector = 2; break;
552 case 0b010: sector = 3; break;
553 case 0b110: sector = 4; break;
554 case 0b100: sector = 5; break;
555 case 0b101: sector = 6; break;
556 default: sector = 0; break; // should not happen
557 }
558 /* SVPWM zero-sequence injection */
559 vmax = h->dutyu_q15;
560 if (h->dutyv_q15 > vmax) vmax = h->dutyv_q15;
561 if (h->dutyw_q15 > vmax) vmax = h->dutyw_q15;
562
563 vmin = h->dutyu_q15;
564 if (h->dutyv_q15 < vmin) vmin = h->dutyv_q15;
565 if (h->dutyw_q15 < vmin) vmin = h->dutyw_q15;
566
567 voff = (vmax + vmin) >> 1;
568
569 h->dutyu_q15 -= voff;
570 h->dutyv_q15 -= voff;
571 h->dutyw_q15 -= voff;
572 mc3p_adc_svm_update(h, sector);
573 eldriver_mc3p_write_phase_duty(h, h->dutyu_q15, h->dutyv_q15, h->dutyw_q15);
574 h->sector_last = sector;
575}
