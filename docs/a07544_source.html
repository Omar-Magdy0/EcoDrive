146void xAEBF(void* argument)
147{
148 while(1)
149 {
150 aebf_message_t msg;
151
152 if(xQueueReceive(aebf_queue_prio1, &msg, 0) == pdPASS)
153 {
154 // Process message
155 switch (msg.type)
156 {
157 case static_cast<uint8_t>(AebfMsgType::Log):
158 // Handle log message (e.g., print to console)
159 memccpy(aebf_encode_buffer.data() + 5, msg.buffer, 0, msg.len);
160 aebf_encode_frame(aebf_encode_buffer.data(), AEBF_DEVICE_ID, AEBF_LOG_SERVICEID, msg.len);
161 break;
162 case static_cast<uint8_t>(AebfMsgType::Debug):
163 // Handle debug message
164 memccpy(aebf_encode_buffer.data() + 5, msg.buffer, 0, msg.len);
165 aebf_encode_frame(aebf_encode_buffer.data(), AEBF_DEVICE_ID, AEBF_DEBUG_SERVICEID, msg.len);
166 break;
167
168 default:
169 break;
170 }
171 serial_write(aebf_encode_buffer.data(), AEBF_FRAMED_SIZE(msg.len));
172 continue;
173 }
174
175 // No messages, check for prio0 aebf_tasks (e.g., PWM stream)
176 PwmDataFrame_t *frame;
177 if(pwmDataBuffer.readFrame(&frame))
178 {
179 // Process a PWM data frame
180 memcpy(aebf_encode_buffer.data() + 5, frame, sizeof(PwmDataFrame_t));
181 aebf_encode_frame(aebf_encode_buffer.data(), AEBF_DEVICE_ID, AEBF_PWM_SERVICEID, sizeof(PwmDataFrame_t));
182 serial_write(aebf_encode_buffer.data(), AEBF_FRAMED_SIZE(sizeof(PwmDataFrame_t)));
183 continue;
184 }
185
186 // If we get here then all messages are processed, wait for notification (either from prioN messages or new PWM data)
187 ulTaskNotifyTake(pdTRUE, // Clear notification count on exit
188 1); // Wait 4ms
189 }
190}
