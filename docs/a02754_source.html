115void eldriver_mc3p_write_svm(eldriver_mc3p_t *h, int16_t alpha_q15, int16_t beta_q15)
116{
117 int32_t vmax, vmin, voff;
118 if(!IS_SVM_SECTOR(h->sector_last))
119 {
120 eldriver_mc3p_write_phase_state(h, ELDRIVER_MC3P_PHASE_COMP, ELDRIVER_MC3P_PHASE_COMP, ELDRIVER_MC3P_PHASE_COMP);
121 h->mode = ELDRIVER_MC3P_MODE_SVM;
122 }
123
124 /* αβ → phase (normalized Q15) */
125 h->dutyu_q15 = alpha_q15;
126
127 h->dutyv_q15 = (-(int32_t)Q15_HALF * alpha_q15
128 + (int32_t)Q15_SQRT3_BY_2 * beta_q15) >> 15;
129
130 h->dutyw_q15 = (-(int32_t)Q15_HALF * alpha_q15
131 - (int32_t)Q15_SQRT3_BY_2 * beta_q15) >> 15;
132
133 uint8_t b0 = (h->dutyu_q15 >= 0);
134 uint8_t b1 = (h->dutyv_q15 >= 0);
135 uint8_t b2 = (h->dutyw_q15 >= 0);
136
137 // 3-bit code
138 uint8_t code = (b2 << 2) | (b1 << 1) | b0;
139 eldriver_mc3p_sector_t sector;
140 switch(code)
141 {
142 case 0b001: sector = 1; break;
143 case 0b011: sector = 2; break;
144 case 0b010: sector = 3; break;
145 case 0b110: sector = 4; break;
146 case 0b100: sector = 5; break;
147 case 0b101: sector = 6; break;
148 default: sector = 0; break; // should not happen
149 }
150
151 /* SVPWM zero-sequence injection */
152 vmax = h->dutyu_q15;
153 if (h->dutyv_q15 > vmax) vmax = h->dutyv_q15;
154 if (h->dutyw_q15 > vmax) vmax = h->dutyw_q15;
155
156 vmin = h->dutyu_q15;
157 if (h->dutyv_q15 < vmin) vmin = h->dutyv_q15;
158 if (h->dutyw_q15 < vmin) vmin = h->dutyw_q15;
159
160 voff = (vmax + vmin) >> 1;
161
162 h->dutyu_q15 -= voff;
163 h->dutyv_q15 -= voff;
164 h->dutyw_q15 -= voff;
165 eldriver_mc3p_write_phase_duty(h, h->dutyu_q15, h->dutyv_q15, h->dutyw_q15);
166 h->sector_last = sector;
167}
