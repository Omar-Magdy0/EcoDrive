23uint8_t hall1_gpioRead()
24{
25 uint8_t state = 0;
26 // Read each pin and shift into the correct bit position
27 if (LL_GPIO_IsInputPinSet(ELDRIVER_HALL1_A_PORT, ELDRIVER_HALL1_A_PIN)) state |= (1 << 2); // MSB
28 if (LL_GPIO_IsInputPinSet(ELDRIVER_HALL1_B_PORT, ELDRIVER_HALL1_B_PIN)) state |= (1 << 1);
29 if (LL_GPIO_IsInputPinSet(ELDRIVER_HALL1_C_PORT, ELDRIVER_HALL1_C_PIN)) state |= (1 << 0); // LSB
30
31 return state;
32}
