39 /* Compute the prescaler value to have TIM11 counter clock equal to 1MHz */
40 uwPrescalerValue = (uint32_t) ((uwTimclock / 1000000U) - 1U);
41
42 /* Initialize TIM11 */
43 htim11.Instance = TIM11;
44
45 /* Initialize TIMx peripheral as follow:
46 * Period = [(TIM11CLK/1000) - 1]. to have a (1/1000) s time base.
47 * Prescaler = (uwTimclock/1000000 - 1) to have a 1MHz counter clock.
48 * ClockDivision = 0
49 * Counter direction = Up
50 */
51 htim11.Init.Period = (1000000U / (ELDRIVER_XMC3P_TICKFREQ)) - 1U;
52 htim11.Init.Prescaler = uwPrescalerValue;
53 htim11.Init.ClockDivision = 0;
54 htim11.Init.CounterMode = TIM_COUNTERMODE_UP;
55 htim11.Init.AutoReloadPreload = TIM_AUTORELOAD_PRELOAD_DISABLE;
56
57 status = HAL_TIM_Base_Init(&htim11);
58 if (status == HAL_OK)
59 {
60 /* Start the TIM time Base generation in interrupt mode */
61 status = HAL_TIM_Base_Start_IT(&htim11);
62 if (status == HAL_OK)
63 {
64 /* Enable the TIM11 global Interrupt */
65 HAL_NVIC_EnableIRQ(TIM1_TRG_COM_TIM11_IRQn);
66 /* Configure the SysTick IRQ priority */
67 if (TickPriority < (1UL << __NVIC_PRIO_BITS))

69 /* Configure the TIM IRQ priority */
70 HAL_NVIC_SetPriority(TIM1_TRG_COM_TIM11_IRQn, TickPriority, 2U);

72 }

75 status = HAL_ERROR;
