135static uint64_t extract_number_and_advance(CborValue *it)
136{
137 /* This function is only called after we've verified that the number
138 * here is valid, so we can just use _cbor_value_extract_int64_helper. */
139 uint8_t descriptor;
140 uint64_t v = _cbor_value_extract_int64_helper(it);
141
142 read_bytes_unchecked(it, &descriptor, 0, 1);
143 descriptor &= SmallValueMask;
144
145 size_t bytesNeeded = descriptor < Value8Bit ? 0 : (1 << (descriptor - Value8Bit));
146 advance_bytes(it, bytesNeeded + 1);
147
148 return v;
149}

157static CborError preparse_value(CborValue *it)
158{
159 enum {
160 /* flags to keep */
161 FlagsToKeep = CborIteratorFlag_ContainerIsMap | CborIteratorFlag_NextIsMapKey
162 };
163 uint8_t descriptor;
164
165 /* are we at the end? */
166 it->type = CborInvalidType;
167 it->flags &= FlagsToKeep;
168 if (!read_bytes(it, &descriptor, 0, 1))
169 return CborErrorUnexpectedEOF;
170
171 uint8_t type = descriptor & MajorTypeMask;
172 it->type = type;
173 it->extra = (descriptor &= SmallValueMask);
174
175 if (descriptor > Value64Bit) {
176 if (unlikely(descriptor != IndefiniteLength))
177 return type == CborSimpleType ? CborErrorUnknownType : CborErrorIllegalNumber;
178 if (likely(!is_fixed_type(type))) {
179 /* special case */
180 it->flags |= CborIteratorFlag_UnknownLength;
181 it->type = type;
182 return CborNoError;
183 }
184 return type == CborSimpleType ? CborErrorUnexpectedBreak : CborErrorIllegalNumber;
185 }
186
187 size_t bytesNeeded = descriptor < Value8Bit ? 0 : (1 << (descriptor - Value8Bit));
188
189 if (bytesNeeded) {
190 if (!can_read_bytes(it, bytesNeeded + 1))
191 return CborErrorUnexpectedEOF;
192
193 it->extra = 0;
194
195 /* read up to 16 bits into it->extra */
196 if (bytesNeeded == 1) {
197 uint8_t extra;
198 read_bytes_unchecked(it, &extra, 1, bytesNeeded);
199 it->extra = extra;
200 } else if (bytesNeeded == 2) {
201 read_bytes_unchecked(it, &it->extra, 1, bytesNeeded);
202 it->extra = cbor_ntohs(it->extra);
203 } else {
204 cbor_static_assert(CborIteratorFlag_IntegerValueTooLarge == (Value32Bit & 3));
205 cbor_static_assert((CborIteratorFlag_IntegerValueIs64Bit |
206 CborIteratorFlag_IntegerValueTooLarge) == (Value64Bit & 3));
207 it->flags |= (descriptor & 3);
208 }
209 }
210
211 uint8_t majortype = type >> MajorTypeShift;
212 if (majortype == NegativeIntegerType) {
213 it->flags |= CborIteratorFlag_NegativeInteger;
214 it->type = CborIntegerType;
215 } else if (majortype == SimpleTypesType) {
216 switch (descriptor) {
217 case FalseValue:
218 it->extra = false;
219 it->type = CborBooleanType;
220 break;
221
222 case SinglePrecisionFloat:
223 case DoublePrecisionFloat:
224 it->flags |= CborIteratorFlag_IntegerValueTooLarge;
225 CBOR_FALLTHROUGH;
226 case TrueValue:
227 case NullValue:
228 case UndefinedValue:
229 case HalfPrecisionFloat:
230 read_bytes_unchecked(it, &it->type, 0, 1);
231 break;
232
233 case SimpleTypeInNextByte:
234#ifndef CBOR_PARSER_NO_STRICT_CHECKS
235 if (unlikely(it->extra < 32)) {
236 it->type = CborInvalidType;
237 return CborErrorIllegalSimpleType;
238 }
239#endif
240 break;
241
242 case 28:
243 case 29:
244 case 30:
245 case Break:
246 cbor_assert(false); /* these conditions can't be reached */
247 return CborErrorUnexpectedBreak;
248 }
249 }
250
251 return CborNoError;
252}

254static CborError preparse_next_value_nodecrement(CborValue *it)
255{
256 uint8_t byte;
257 if (it->remaining == UINT32_MAX && read_bytes(it, &byte, 0, 1) && byte == (uint8_t)BreakByte) {
258 /* end of map or array */
259 if ((it->flags & CborIteratorFlag_ContainerIsMap && it->flags & CborIteratorFlag_NextIsMapKey)
260 || it->type == CborTagType) {
261 /* but we weren't expecting it! */
262 return CborErrorUnexpectedBreak;
263 }
264 it->type = CborInvalidType;
265 it->remaining = 0;
266 it->flags |= CborIteratorFlag_UnknownLength; /* leave_container must consume the Break */
267 return CborNoError;
268 }
269
270 return preparse_value(it);
271}

273static CborError preparse_next_value(CborValue *it)
274{
275 /* tags don't count towards item totals or whether we've successfully
276 * read a map's key or value */
277 bool itemCounts = it->type != CborTagType;
278
279 if (it->remaining != UINT32_MAX) {
280 if (itemCounts && --it->remaining == 0) {
281 it->type = CborInvalidType;
282 it->flags &= ~CborIteratorFlag_UnknownLength; /* no Break to consume */
283 return CborNoError;
284 }
285 }
286 if (itemCounts) {
287 /* toggle the flag indicating whether this was a map key */
288 it->flags ^= CborIteratorFlag_NextIsMapKey;
289 }
290 return preparse_next_value_nodecrement(it);
291}

293static CborError advance_internal(CborValue *it)
294{
295 uint64_t length = extract_number_and_advance(it);
296
297 if (it->type == CborByteStringType || it->type == CborTextStringType) {
298 cbor_assert(length == (size_t)length);
299 cbor_assert((it->flags & CborIteratorFlag_UnknownLength) == 0);
300 advance_bytes(it, length);
301 }
302
303 return preparse_next_value(it);
304}

463static CborError advance_recursive(CborValue *it, int nestingLevel)
464{
465 CborError err;
466 CborValue recursed;
467
468 if (is_fixed_type(it->type))
469 return advance_internal(it);
470
471 if (!cbor_value_is_container(it)) {
472 size_t len = SIZE_MAX;
473 return _cbor_value_copy_string(it, NULL, &len, it);
474 }
475
476 /* map or array */
477 if (nestingLevel == 0)
478 return CborErrorNestingTooDeep;
479
480 err = cbor_value_enter_container(it, &recursed);
481 if (err)
482 return err;
483 while (!cbor_value_at_end(&recursed)) {
484 err = advance_recursive(&recursed, nestingLevel - 1);
485 if (err)
486 return err;
487 }
488 return cbor_value_leave_container(it, &recursed);
489}

569CborError cbor_value_enter_container(const CborValue *it, CborValue *recursed)
570{
571 cbor_static_assert(CborIteratorFlag_ContainerIsMap == (CborMapType & ~CborArrayType));
572 cbor_assert(cbor_value_is_container(it));
573 *recursed = *it;
574
575 if (it->flags & CborIteratorFlag_UnknownLength) {
576 recursed->remaining = UINT32_MAX;
577 advance_bytes(recursed, 1);
578 } else {
579 uint64_t len = extract_number_and_advance(recursed);
580
581 recursed->remaining = (uint32_t)len;
582 if (recursed->remaining != len || len == UINT32_MAX) {
583 /* back track the pointer to indicate where the error occurred */
584 copy_current_position(recursed, it);
585 return CborErrorDataTooLarge;
586 }
587 if (recursed->type == CborMapType) {
588 /* maps have keys and values, so we need to multiply by 2 */
589 if (recursed->remaining > UINT32_MAX / 2) {
590 /* back track the pointer to indicate where the error occurred */
591 copy_current_position(recursed, it);
592 return CborErrorDataTooLarge;
593 }
594 recursed->remaining *= 2;
595 }
596 if (len == 0) {
597 /* the case of the empty container */
598 recursed->type = CborInvalidType;
599 return CborNoError;
600 }
601 }
602 recursed->flags = (recursed->type & CborIteratorFlag_ContainerIsMap);
603 return preparse_next_value_nodecrement(recursed);
604}
605
618CborError cbor_value_leave_container(CborValue *it, const CborValue *recursed)
619{
620 cbor_assert(cbor_value_is_container(it));
621 cbor_assert(recursed->type == CborInvalidType);
622
623 copy_current_position(it, recursed);
624 if (recursed->flags & CborIteratorFlag_UnknownLength)
625 advance_bytes(it, 1);
626 return preparse_next_value(it);
627}

803CborError cbor_value_get_int64_checked(const CborValue *value, int64_t *result)
804{
805 uint64_t v;
806 cbor_assert(cbor_value_is_integer(value));
807 v = _cbor_value_extract_int64_helper(value);
808
809 /* Check before converting, as the standard says (C11 6.3.1.3 paragraph 3):
810 * "[if] the new type is signed and the value cannot be represented in it; either the
811 * result is implementation-defined or an implementation-defined signal is raised."
812 *
813 * The range for int64_t is -2^63 to 2^63-1 (int64_t is required to be
814 * two's complement, C11 7.20.1.1 paragraph 3), which in CBOR is
815 * represented the same way, differing only on the "sign bit" (the major
816 * type).
817 */
818
819 if (unlikely(v > (uint64_t)INT64_MAX))
820 return CborErrorDataTooLarge;
821
822 *result = v;
823 if (value->flags & CborIteratorFlag_NegativeInteger)
824 *result = -*result - 1;
825 return CborNoError;
826}

843CborError cbor_value_get_int_checked(const CborValue *value, int *result)
844{
845 uint64_t v;
846 cbor_assert(cbor_value_is_integer(value));
847 v = _cbor_value_extract_int64_helper(value);
848
849 /* Check before converting, as the standard says (C11 6.3.1.3 paragraph 3):
850 * "[if] the new type is signed and the value cannot be represented in it; either the
851 * result is implementation-defined or an implementation-defined signal is raised."
852 *
853 * But we can convert from signed to unsigned without fault (paragraph 2).
854 *
855 * The range for int is implementation-defined and int is not guaranteed to use
856 * two's complement representation (although int32_t is).
857 */
858
859 if (value->flags & CborIteratorFlag_NegativeInteger) {
860 if (unlikely(v > (unsigned) -(INT_MIN + 1)))
861 return CborErrorDataTooLarge;
862
863 *result = (int)v;
864 *result = -*result - 1;
865 } else {
866 if (unlikely(v > (uint64_t)INT_MAX))
867 return CborErrorDataTooLarge;
868
869 *result = (int)v;
870 }
871 return CborNoError;
872
873}

981static CborError get_string_chunk_size(const CborValue *it, size_t *offset, size_t *len)
982{
983 uint8_t descriptor;
984 size_t bytesNeeded = 1;
985
986 if (cbor_value_is_length_known(it) && (it->flags & CborIteratorFlag_BeforeFirstStringChunk) == 0)
987 return CborErrorNoMoreStringChunks;
988
989 /* are we at the end? */
990 if (!read_bytes(it, &descriptor, 0, 1))
991 return CborErrorUnexpectedEOF;
992
993 if (descriptor == BreakByte)
994 return CborErrorNoMoreStringChunks;
995 if ((descriptor & MajorTypeMask) != it->type)
996 return CborErrorIllegalType;
997
998 /* find the string length */
999 descriptor &= SmallValueMask;
1000 if (descriptor < Value8Bit) {
1001 *len = descriptor;
1002 } else if (unlikely(descriptor > Value64Bit)) {
1003 return CborErrorIllegalNumber;
1004 } else {
1005 uint64_t val;
1006 bytesNeeded = (size_t)(1 << (descriptor - Value8Bit));
1007 if (!can_read_bytes(it, 1 + bytesNeeded))
1008 return CborErrorUnexpectedEOF;
1009
1010 if (descriptor <= Value16Bit) {
1011 if (descriptor == Value16Bit)
1012 val = read_uint16(it, 1);
1013 else
1014 val = read_uint8(it, 1);
1015 } else {
1016 if (descriptor == Value32Bit)
1017 val = read_uint32(it, 1);
1018 else
1019 val = read_uint64(it, 1);
1020 }
1021
1022 *len = val;
1023 if (*len != val)
1024 return CborErrorDataTooLarge;
1025
1026 ++bytesNeeded;
1027 }
1028
1029 *offset = bytesNeeded;
1030 return CborNoError;
1031}

1039static CborError get_string_chunk(CborValue *it, const void **bufferptr, size_t *len)
1040{
1041 size_t offset;
1042 CborError err = get_string_chunk_size(it, &offset, len);
1043 if (err)
1044 return err;
1045
1046 /* we're good, transfer the string now */
1047 err = transfer_string(it, bufferptr, offset, *len);
1048 if (err)
1049 return err;
1050
1051 /* we've iterated at least once */
1052 it->flags &= ~CborIteratorFlag_BeforeFirstStringChunk;
1053 return CborNoError;
1054}

1147/* We return uintptr_t so that we can pass memcpy directly as the iteration
1148 * function. The choice is to optimize for memcpy, which is used in the base
1149 * parser API (cbor_value_copy_string), while memcmp is used in convenience API
1150 * only. */

1156 (void)src;
1157 (void)len;
1158 return true;
1159}

1171static CborError iterate_string_chunks(const CborValue *value, char *buffer, size_t *buflen,
1172 bool *result, CborValue *next, IterateFunction func)
1173{
1174 CborError err;
1175 CborValue tmp;
1176 size_t total = 0;
1177 const void *ptr;
1178
1179 cbor_assert(cbor_value_is_byte_string(value) || cbor_value_is_text_string(value));
1180 if (!next)
1181 next = &tmp;
1182 *next = *value;
1183 *result = true;
1184
1185 err = _cbor_value_begin_string_iteration(next);
1186 if (err)
1187 return err;
1188
1189 while (1) {
1190 size_t newTotal;
1191 size_t chunkLen;
1192 err = get_string_chunk(next, &ptr, &chunkLen);
1193 if (err == CborErrorNoMoreStringChunks)
1194 break;
1195 if (err)
1196 return err;
1197
1198 if (unlikely(add_check_overflow(total, chunkLen, &newTotal)))
1199 return CborErrorDataTooLarge;
1200
1201 if (*result && *buflen >= newTotal)
1202 *result = !!func(buffer == NULL ? buffer : buffer + total, (const uint8_t *)ptr, chunkLen);
1203 else
1204 *result = false;
1205
1206 total = newTotal;
1207 }
1208
1209 /* is there enough room for the ending NUL byte? */
1210 if (*result && *buflen > total) {
1211 uint8_t nul[] = { 0 };
1212 *result = !!func(buffer == NULL ? buffer : buffer + total, nul, 1);
1213 }
1214 *buflen = total;
1215 return _cbor_value_finish_string_iteration(next);
1216}

1286CborError _cbor_value_copy_string(const CborValue *value, void *buffer,
1287 size_t *buflen, CborValue *next)

1402CborError cbor_value_map_find_value(const CborValue *map, const char *string, CborValue *element)
1403{
1404 CborError err;
1405 size_t len = strlen(string);
1406 cbor_assert(cbor_value_is_map(map));
1407 err = cbor_value_enter_container(map, element);
1408 if (err)
1409 goto error;
1410
1411 while (!cbor_value_at_end(element)) {
1412 /* find the non-tag so we can compare */
1413 err = cbor_value_skip_tag(element);
1414 if (err)
1415 goto error;
1416 if (cbor_value_is_text_string(element)) {
1417 bool equals;
1418 size_t dummyLen = len;
1419 err = iterate_string_chunks(element, CONST_CAST(char *, string), &dummyLen,
1420 &equals, element, iterate_memcmp);
1421 if (err)
1422 goto error;
1423 if (equals)
1424 return preparse_value(element);
1425 } else {
1426 /* skip this key */
1427 err = cbor_value_advance(element);
1428 if (err)
1429 goto error;
1430 }
1431
1432 /* skip this value */
1433 err = cbor_value_skip_tag(element);
1434 if (err)
1435 goto error;
1436 err = cbor_value_advance(element);
1437 if (err)
1438 goto error;
1439 }
1440
1441 /* not found */
1442 element->type = CborInvalidType;
1443 return CborNoError;
1444
1445error:
1446 element->type = CborInvalidType;
1447 return err;
1448}
