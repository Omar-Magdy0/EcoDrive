158void pwmDataBuffer_t::pushSample()
159{
160 if (frame_sample_idx < SAMPLES_PER_FRAME)
161 {
162 frame_sample_idx++;
163 }
164 sample_count++;
165 if (frame_sample_idx >= SAMPLES_PER_FRAME)
166 {
167 elcore_rstream_commitWrite(&buffer, 1); // Commit the current frame
168 // Current frame is full, move to next frame
169 uint8_t *w2;
170 uint16_t c1, c2;
171 // Commit the current full frame
172 if (elcore_rstream_reserveWrite(&buffer, 1, (void **)&currentFrame, &c1, (void **)&w2, &c2))
173 {
174 currentFrame->sample_counter = sample_count;
175 frame_sample_idx = 0;
176 }
177 else
178 {
179 // Buffer overflow, data loss occurs
